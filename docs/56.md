type AuthenticationFactorConnection {
  """A list of the entities provided by this connection."""
  entities: [AuthenticationFactor]!

  """
  An object providing information about the page of results being displayed, as
  well as the total amount of pages/records available.
  """
  page_info: PageInfo!

  """
  Provides the ability to return aggregated mathematical data about your results.
  """
  aggregations: [Aggregation]!
}

"""The authentication factor type."""
enum AuthenticationFactorType {
  """Receive one-time passcodes via email"""
  EMAIL

  """Receive one-time passcodes via SMS"""
  SMS

  """
  Generate time-based one-time passcodes with a user-configured application
  """
  TOTP

  """
  Generate a set of single-use codes for use when no other authentication factors are available
  """
  RECOVERY_CODES
}
