```graphql
type CustomFieldDataConnection {
  """A list of the entities provided by this connection."""
  entities: [CustomFieldData]!

  """
  An object providing information about the page of results being displayed, as
  well as the total amount of pages/records available.
  """
  page_info: PageInfo!

  """
  Provides the ability to return aggregated mathematical data about your results.
  """
  aggregations: [Aggregation]!
}

"""Allows the addition of custom field data to the parent entity."""
input CustomFieldDataMutationInput {
  """The ID of a CustomField that is associated with this type of entity."""
  custom_field_id: Int64Bit!

  """
  The value of the custom field. For custom fields of the TEXT type, this should
  be arbitrary text, or one of the allowed values. For custom fields of the DATE
  type, this should be a date in YYYY-MM-DD format. For custom fields of the
  BOOLEAN type, this should be 1 or 0.
  """
  value: String!
}

"""Different types of `CustomField`s."""
enum CustomFieldType {
  """A boolean value"""
  BOOLEAN

  """A date input"""
  DATE

  """A text input"""
  TEXT
}

"""A user defined link."""
```
