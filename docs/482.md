type SignatureConnection {
  """A list of the entities provided by this connection."""
  entities: [Signature]!

  """
  An object providing information about the page of results being displayed, as
  well as the total amount of pages/records available.
  """
  page_info: PageInfo!

  """
  Provides the ability to return aggregated mathematical data about your results.
  """
  aggregations: [Aggregation]!
}

"""An entity that can have an SMS message sent to it."""
interface SmsableInterface {
  """The ID of the entity."""
  id: Int64Bit!

  """
  An ID that uniquely identifies this entity across the whole Sonar system.
  """
  sonar_unique_id: ID!

  """The date and time this entity was created."""
  created_at: Datetime!

  """The last date and time this entity was modified."""
  updated_at: Datetime!
}

"""The entity that this `SMS` is associated with."""
enum SmsableType {
  """A `User`."""
  User

  """A `Contact`."""
  Contact
}

"""An SMS contact prefix message."""
enum SmsContactPrefix {
